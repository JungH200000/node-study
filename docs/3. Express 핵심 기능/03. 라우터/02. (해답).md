## 내가 작성한 답안

```js
// /node03-express-core-features
// /src/app.js
import express from 'express';

const app = express();
const PORT = 3000;

app
  .route('/users')
  .get((req, res) => {
    res.json({ message: 'User 목록 보기' });
  })
  .post((req, res) => {
    res.json({ message: 'User 추가하기' });
  });

app.listen(PORT, () => {
  console.log(`🚀 Server started on port ${PORT}`);
});
```

## 모범 답안

```js
// /node03-express-core-features
// /src/app.js
import express from 'express';

const app = express();
const PORT = 3000;

    res.json({ message: 'User 추가하기' });
  });

app.listen(PORT, () => {
  console.log(`🚀 Server started on port ${PORT}`);
});
```
